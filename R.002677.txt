SELECT branchname, time_queued, to_address, status, message_id, content
FROM message_queue 
LEFT JOIN branches ON branchemail = to_address
WHERE date(time_queued) = CURDATE() AND branchname IS NOT NULL AND to_address <> '' AND letter_code=''
ORDER BY branchcode
LIMIT 100
