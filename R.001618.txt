SELECT p.surname, p.firstname, p.cardnumber, p.dateexpiry, p.dateenrolled, p.branchcode, p.address, p.city, p.phone, p.email, p.borrowernotes,m.message,a.amountoutstanding
FROM borrowers p
LEFT JOIN accountlines a USING(borrowernumber) 
LEFT JOIN messages m USING(borrowernumber) 
WHERE borrowernumber IN (SELECT borrowernumber FROM issues) AND
(a.amountoutstanding !='0.00' OR a.amountoutstanding IS NOT
NULL) AND p.categorycode= <<Patron Category|categorycode>> 
GROUP BY p.borrowernumber
