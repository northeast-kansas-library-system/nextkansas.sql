SELECT
  items.itemnumber,
  items.barcode
FROM
  items LEFT JOIN
  (SELECT
    branchtransfers.itemnumber,
    branchtransfers.datesent,
    branchtransfers.datearrived
  FROM
    branchtransfers
  WHERE
    branchtransfers.datearrived IS NULL) transferss
    ON items.itemnumber = transferss.itemnumber LEFT JOIN
  (SELECT
    reserves.itemnumber,
    reserves.waitingdate
  FROM
    reserves
  WHERE
    reserves.waitingdate IS NOT NULL) reservess
    ON items.itemnumber = reservess.itemnumber
WHERE
  items.homebranch = 'ATCHISON' AND
  Coalesce(items.itemlost, 0) = 0 AND
  Coalesce(items.withdrawn, 0) = 0 AND
  Coalesce(transferss.datesent, "-") LIKE "-" AND
  Coalesce(reservess.waitingdate, "-") LIKE "-"
GROUP BY
  items.itemnumber, items.barcode, items.homebranch, Coalesce(items.itemlost,
  0), Coalesce(items.withdrawn, 0), Coalesce(transferss.datesent, "-"),
  Coalesce(reservess.waitingdate, "-")
