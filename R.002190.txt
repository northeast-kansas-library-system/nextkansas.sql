Written for libraries to edit records with itemnotes data (public note for an item). Ordered by dateaccessioned, desc. Includes direct edit item link.

----------

SELECT CONCAT ('<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',b.biblionumber,'\" target="_blank">'"link to record"'</a>') as "link to record", i.dateaccessioned, b.title, b.author, i.itemnotes, CONCAT('<a href=\"/cgi-bin/koha/cataloguing/additem.pl?op=edititem&biblionumber=',b.biblionumber,'&itemnumber=',i.itemnumber,'#edititem\" target="_blank">'"edit item"'</a>') as "edit item"  FROM items i JOIN biblio b USING (biblionumber) WHERE i.homebranch=<<Pick your branch|branches>> AND i.itemnotes IS NOT NULL ORDER BY i.dateaccessioned DESC
