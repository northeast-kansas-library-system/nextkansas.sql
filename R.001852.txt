SELECT statistics.datetime, borrowers.borrowernumber, CONCAT (borrowers.firstname, ' ', borrowers.surname) AS 'name', statistics.type, borrowers.cardnumber, biblio.title, biblio.biblionumber, items.itemnumber, items.barcode, items.itype FROM borrowers LEFT JOIN statistics USING(borrowernumber) LEFT JOIN items USING(itemnumber) LEFT JOIN biblio USING(biblionumber) WHERE MONTH(datetime)='01' AND YEAR(datetime)='2013' AND statistics.type IN ('issue','renew','localuse') AND statistics.branch='HORTON' AND items.itype='COMPUTER' ORDER BY statistics.datetime DESC LIMIT 5000
