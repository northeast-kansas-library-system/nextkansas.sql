SELECT
  If(deletedbiblio.biblionumber IS NULL, "Y", "") AS BIBLIO_DELETED,
  Coalesce(biblio.biblionumber, deletedbiblio.biblionumber) AS biblionumber,
  deleteditems.itemnumber,
  deleteditems.barcode,
  deleteditems.homebranch,
  deleteditems.location,
  deleteditems.itype,
  deleteditems.ccode,
  deleteditems.itemcallnumber,
  Coalesce(biblio.author, deletedbiblio.author) AUTHOR,
  Coalesce(biblio.title, deletedbiblio.title) AS TITLE,
  outstanding.amountoutstanding
FROM
  deleteditems
  LEFT JOIN biblio
    ON biblio.biblionumber = deleteditems.biblionumber
  LEFT JOIN deletedbiblio
    ON deletedbiblio.biblionumber = deleteditems.biblionumber
  LEFT JOIN (
    SELECT
      accountlines.description,
      Right(accountlines.description, 14) AS BC,
      accountlines.amountoutstanding
    FROM
      accountlines
    WHERE
      accountlines.description LIKE "%ILEA%"
    LIMIT 5000
  ) outstanding
    ON outstanding.BC = deleteditems.barcode
WHERE
  deleteditems.homebranch = 'LEAVENWRTH'
GROUP BY
  deleteditems.itemnumber
LIMIT 5000
