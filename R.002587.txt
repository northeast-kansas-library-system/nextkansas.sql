SELECT branches.branchcode as Library, 
(SELECT count(*) FROM branchtransfers LEFT JOIN items USING (itemnumber) WHERE branches.branchcode=items.homebranch AND (items.homebranch != branchtransfers.tobranch) AND (branchtransfers.frombranch != branchtransfers.tobranch) AND MONTH(branchtransfers.datesent)='08' AND YEAR(branchtransfers.datesent)='2015') as loans,
(SELECT count(*) FROM branchtransfers LEFT JOIN items USING(itemnumber) WHERE branches.branchcode=branchtransfers.tobranch AND (branchtransfers.tobranch != items.homebranch) AND (branchtransfers.tobranch != branchtransfers.frombranch) AND MONTH(branchtransfers.datesent)='08' AND YEAR(branchtransfers.datesent)='2015') as borrows
FROM branches WHERE branches.branchcode ='OSKALOOSA'
GROUP BY branches.branchcode 
LIMIT 100
