SELECT
  virtualshelves.shelfname,
  virtualshelfcontents.biblionumber,
  biblio.biblionumber AS biblionumber1,
  biblio.author,
  biblio.title
FROM
  virtualshelves
  JOIN virtualshelfcontents ON virtualshelfcontents.shelfnumber = virtualshelves.shelfnumber
  JOIN biblio ON virtualshelfcontents.biblionumber = biblio.biblionumber
WHERE
  virtualshelves.shelfname = <<Choose|virtualshelves.shelfname>>
GROUP BY
  virtualshelves.shelfname,
  virtualshelfcontents.biblionumber,
  biblio.biblionumber,
  biblio.author,
  biblio.title
