SELECT statistics.branch as Library, biblio.title, biblio.author, items.itemcallnumber, items.itype, items.ccode, items.location, items.homebranch as ItemBranch, statistics.itemtype FROM statistics  LEFT JOIN items USING (itemnumber) LEFT JOIN biblio USING(biblionumber) WHERE statistics.branch = <<Pick your branch|branches>> AND YEAR(statistics.datetime)=<<YEAR (yyyy format, ex 2014, enter 2014)>> AND MONTH(statistics.datetime)= <<Month (mm format, ex June, enter 06)>> AND statistics.branch != items.homebranch AND statistics.type IN ('issue', 'renew') ORDER BY statistics.branch, biblio.title
