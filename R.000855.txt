Select branch, enter call number prefix, and run

----------

SELECT items.itype,items.itemcallnumber,items.barcode,biblio.title,biblioitems.publicationyear,items.dateaccessioned FROM items LEFT JOIN biblio USING (biblionumber) LEFT JOIN biblioitems USING(biblionumber) WHERE items.homebranch=<<branch|branches>> AND items.itemcallnumber LIKE <<Enter Call Number prefix %>> ORDER BY items.dateaccessioned DESC
