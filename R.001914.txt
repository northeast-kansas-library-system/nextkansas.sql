SELECT surname AS Surname, firstname AS "First Name", cardnumber AS
"Card Number", email AS Email
FROM borrowers
WHERE email NOT LIKE '%_@__%.__%' AND branchcode=<<choose branch|branches>>
ORDER BY email DESC
