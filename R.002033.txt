New items added to your collection in the last 24 hours. This report is limited to 300 items. Clicking on the linked record number will take you to that record

----------

SELECT CONCAT('<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',biblio.biblionumber,'\" target="_blank">',biblio.biblionumber,'</a>') AS "Link to Record", items.dateaccessioned, biblio.author, biblio.title, items.itemcallnumber, items.ccode, items.itype, items.location, items.barcode, items.price, items.replacementprice, items.timestamp FROM items JOIN biblio USING(biblionumber) WHERE items.homebranch=<<Pick your branch|branches>> AND items.dateaccessioned >= DATE_SUB(NOW(), INTERVAL 1 DAY)
ORDER BY items.dateaccessioned DESC LIMIT 300
