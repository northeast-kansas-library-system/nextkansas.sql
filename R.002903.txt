SELECT
  borrowers.branchcode,
  borrowers.categorycode,
  COUNT(*) AS P_ADDED_LM
FROM
  borrowers
WHERE
  Month(borrowers.dateenrolled) = Month(Now() - INTERVAL 1 MONTH) AND
  Year(borrowers.dateenrolled) = Year(Now() - INTERVAL 1 MONTH) AND
  borrowers.branchcode LIKE <<Choose patron home library|LBRANCH>>
GROUP BY
  borrowers.branchcode, borrowers.categorycode WITH ROLLUP
