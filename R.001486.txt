select CONCAT('<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',b.biblionumber,'\" target="_blank">',b.biblionumber,'</a>') as 'Bib Number', b.title, ExtractValue(t.marcxml, '//datafield[@tag="245"]/subfield[@code>="b"]') as 'Remainder of Title', ExtractValue(t.marcxml, '//datafield[@tag="250"]/subfield[@code>="a"]') as Edition, ExtractValue(t.marcxml, '//datafield[@tag="500"]/subfield[@code>="a"]') as 'General Note', ExtractValue(t.marcxml, '//datafield[@tag="505"]/subfield[@code>="a"]') as Contents, ExtractValue(t.marcxml, '//datafield[@tag="538"]/subfield[@code>="a"]') as 'Sys req', i.itemcallnumber, t.pages, t.illus, t.size, i.homebranch, i.barcode FROM biblio b LEFT JOIN items i USING (biblionumber) LEFT JOIN biblioitems t USING (biblionumber) WHERE i.ccode='VIDEOGAME'
