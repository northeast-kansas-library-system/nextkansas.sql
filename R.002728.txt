SELECT
  borrowers.borrowernumber,
  borrowers.branchcode,
  borrowers.cardnumber,
  borrowers.categorycode,
  borrower_attribute_types.authorised_value_category,
  borrower_attributes.attribute
FROM
  borrowers LEFT JOIN
  borrower_attributes
    ON borrower_attributes.borrowernumber = borrowers.borrowernumber LEFT JOIN
  borrower_attribute_types
    ON borrower_attributes.code = borrower_attribute_types.code
WHERE
  borrowers.branchcode LIKE 'PH%' AND
  borrower_attribute_types.authorised_value_category = 'GRADYEAR'
