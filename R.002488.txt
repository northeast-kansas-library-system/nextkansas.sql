SELECT borrowers.firstname, borrowers.surname, borrowers.address as "school", biblio.title, items.ccode, items.barcode, items.homebranch as "owning library", issues.date_due, issues.issuedate, issues.lastreneweddate  FROM biblio LEFT JOIN items USING(biblionumber) LEFT JOIN issues USING(itemnumber) LEFT JOIN borrowers USING(borrowernumber) WHERE borrowernumber IN (SELECT borrowernumber FROM borrowers WHERE branchcode='OSAWATOMIE' AND (address LIKE '%OHS%' OR address LIKE '%OMS%' OR address LIKE '%Trojan%' OR address LIKE '%Swenson%')) ORDER BY address, surname, firstname
