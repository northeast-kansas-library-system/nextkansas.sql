SELECT
  CONCAT(
    borrowers.firstname,
    " ",
    borrowers.surname,
    "<br />",
    borrowers.address,
    "<br />",
    borrowers.city,
    ", ",
    borrowers.state,
    " ",
    borrowers.zipcode)  AS SINGLE_ADDRESS_LABEL  
FROM
  borrowers
WHERE
  borrowers.cardnumber LIKE Concat("%", <<Enter patron barcode number>>, "%")
