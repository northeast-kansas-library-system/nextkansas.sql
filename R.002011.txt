SELECT CONCAT('<a href=\"/cgi-bin/koha/members/memberentry.pl?op=modify&borrowernumber=',b.borrowernumber,'\" target="_blank">'"edit patron"'</a>') as "edit patron", b.borrowernumber,  b.cardnumber, b.surname, b.firstname, b.othernames, b.email, b.dateofbirth, b.branchcode, b.categorycode, b.userid, b.sort1,b.password,
ba_GRADE.attribute AS grade, ba_GRADYR.attribute AS grad_year, ba_OLDKEY.attribute AS old_key, ba_UPDATED.attribute AS updated
FROM borrowers b
LEFT JOIN borrower_attributes ba_GRADE ON (b.borrowernumber = ba_GRADE.borrowernumber AND ba_GRADE.code = 'GRADE')
LEFT JOIN borrower_attributes ba_GRADYR ON (b.borrowernumber = ba_GRADYR.borrowernumber AND ba_GRADYR.code = 'GRADYR')
LEFT JOIN borrower_attributes ba_OLDKEY ON (b.borrowernumber = ba_OLDKEY .borrowernumber AND ba_OLDKEY .code = 'OLDKEY ')
LEFT JOIN borrower_attributes ba_UPDATED ON (b.borrowernumber = ba_UPDATED .borrowernumber AND ba_UPDATED .code = 'UPDATED ')
WHERE b.branchcode IN ('PHAXTELL','PHSUMM','phshs','phses','phsms','phwac') AND b.categorycode='STUDENT' AND b.dateofbirth IS NULL
GROUP BY b.borrowernumber
ORDER BY b.surname,  b.firstname
LIMIT 2000
