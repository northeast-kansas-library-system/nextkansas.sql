SELECT
  action_logs.action_id,
  action_logs.timestamp,
  action_logs.user,
  action_logs.module,
  action_logs.action,
  action_logs.object,
  replace(replace(action_logs.info, CHAR(13), 'ZAZAZ '), CHAR(10), 'YAYAY')
FROM
  action_logs
WHERE
  action_logs.timestamp > Now() - INTERVAL 10 DAY AND
  action_logs.module = 'SYSTEMPREFERENCE'
