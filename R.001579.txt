SELECT count(i.barcode), i.homebranch FROM items i WHERE i.itype IN ('LOCALHOLD','LOCALHOLD1','LOCALHOLD2') AND  i.dateaccessioned >= <<Date you want to count (example: 2011-11-16) >> AND i.dateaccessioned < <<Day after the Date you want to count (example: 2011-11-17)>> AND i.ccode <> 'MAGAZINE' AND i.ccode <> 'COMPUTER' AND i.ccode <> 'GADGET' GROUP BY i.homebranch ORDER BY i.homebranch
