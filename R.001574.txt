SELECT i.barcode, i.itemcallnumber, i.ccode, i.location, b.title, b.author, t.publicationyear FROM biblio b LEFT JOIN items i USING(biblionumber) LEFT JOIN biblioitems t USING(biblionumber) WHERE i.biblionumber IN (SELECT biblionumber FROM items WHERE homebranch=<<branch|branches>> AND ccode=<<ccode|CCODE>> GROUP BY biblionumber HAVING count(itemnumber) > 1) AND i.homebranch=<<same branch|branches>> AND i.ccode =<<same ccode|CCODE>>
