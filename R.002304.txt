SELECT CONCAT('<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',b.biblionumber,'\" target="_blank">',b.biblionumber,'</a>') AS "bib number", i.holdingbranch, i.barcode as 'Barcode', i.itemcallnumber, b.title, t.publicationyear as 'Copyright',i.dateaccessioned as 'Accessioned', i.itype, i.ccode, i.issues, i.renewals, (IFNULL(i.issues,0)+IFNULL(i.renewals,0)) as Total_Circ, i.datelastborrowed, i.itemlost, i.onloan, i.damaged,i.itemnotes FROM items i LEFT JOIN biblio b USING (biblionumber) LEFT JOIN biblioitems t USING(biblionumber) WHERE  i.homebranch='valleyfall' AND  i.itype NOT IN ('MEDIA','AUDIOBOOK') ORDER BY i.itemcallnumber
