SELECT b.title, GROUP_CONCAT(i.itemcallnumber ORDER BY i.itemcallnumber ASC SEPARATOR ', ') as callnumbers, t.editionstatement as edition, t.pages, ExtractValue(t.marcxml, '//datafield[@tag="538"]/subfield[@code>="a"]') as sysdetails, CONCAT ('<a href="https://catalog.nexpresslibrary.org/cgi-bin/koha/opac-detail.pl?biblionumber=',b.biblionumber,'\" target="_blank">',b.biblionumber,'</a>') AS "Link to opac record", GROUP_CONCAT(ta.term ORDER BY ta.term ASC SEPARATOR ', ') as tags
FROM biblio b LEFT JOIN items i USING(biblionumber) LEFT JOIN biblioitems t USING(biblionumber) LEFT JOIN tags_all ta USING(biblionumber)
WHERE i.ccode='VIDEOGAME' GROUP BY b.biblionumber
ORDER BY tags
