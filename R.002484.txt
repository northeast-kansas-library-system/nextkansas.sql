SELECT cardnumber, surname, firstname,debarred, dateexpiry,FORMAT(SUM(amountoutstanding),2) as money_owed, (SELECT count(issues.itemnumber) FROM issues WHERE issues.borrowernumber=borrowers.borrowernumber) as items_checked_out,
CONCAT( '<a href=\"/cgi-bin/koha/circ/circulation.pl?borrowernumber=',borrowernumber ,'\" target="_blank">', borrowernumber, '</a>' ) AS 'LINK_TO_PATRON',
CONCAT( '<a href=\"/cgi-bin/koha/members/pay.pl?borrowernumber=',borrowernumber ,'\" target="_blank">', borrowernumber, '</a>' ) AS 'LINK_TO_PATRON_FEES'
FROM borrowers LEFT JOIN accountlines USING(borrowernumber) WHERE borrowers.branchcode='LEAVENWRTH' AND cardnumber LIKE 'LEAVRESET%' GROUP BY borrowers.cardnumber ORDER BY dateexpiry, surname, firstname
