SELECT
  biblio_metadata.biblionumber,
  CONCAT("<a href='/cgi-bin/koha/cataloguing/editor.pl#catalog/", biblio_metadata.biblionumber, "' target='_blank'>Go to record</a>") AS LINK
FROM
  biblio_metadata
WHERE
  Hex(biblio_metadata.metadata) LIKE "%efbfbd%"
ORDER BY
  biblio_metadata.biblionumber
  DESC
