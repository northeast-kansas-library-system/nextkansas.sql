select i.timestamp, CONCAT('<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',b.biblionumber,'\" target="_blank">',b.biblionumber,'</a>') as "Linked Record", CONCAT('<a href=\"/cgi-bin/koha/catalogue/search.pl?idx=ti&q=',b.title,'&sort_by=title_az\" target="_blank">',b.title,'</a>') AS "Title Search", i.barcode, i.itemcallnumber, i.ccode, i.itype, i.location, i.replacementprice, i.notforloan, i.homebranch, i.holdingbranch, i.dateaccessioned, t.pages, t.size, t.isbn from biblio b join items i using (biblionumber) join biblioitems t using (biblionumber) where i.homebranch=<<branch|branches>> AND i.ccode=<<ccode|CCODE>> order by b.biblionumber DESC LIMIT 10000
