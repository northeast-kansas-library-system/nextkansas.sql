SELECT
  accountlines.accountno,
  borrowers.borrowernumber,
  round(Sum(accountlines.amount),2) AS 'Fines Paid',
  CONCAT('<a href=\"/cgi-bin/koha/members/boraccount.pl?borrowernumber=',borrowers.borrowernumber,'" target="_blank">',borrowers.cardnumber,'</a>') as "link to borrower",
  borrowers.firstname,
  borrowers.surname,
  biblio.title,
  accountlines.description,
  accountlines.accounttype,
  accountlines.timestamp as 'paid date'
FROM accountlines
  LEFT JOIN borrowers USING (borrowernumber)
  LEFT join items using (itemnumber)
  LEFT join biblio using(biblionumber)
WHERE borrowers.branchcode= <<Pick your branch|branches>>
GROUP BY accountlines.timestamp
