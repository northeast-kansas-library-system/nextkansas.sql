SELECT
  items.itemnumber,
  items.permanent_location,
  items.location
FROM
  items
WHERE
  (items.permanent_location = 'BALDYA') OR
  (items.location = 'BALDYA')
GROUP BY
  items.itemnumber,
  items.permanent_location,
  items.location
LIMIT 1000
