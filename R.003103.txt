SELECT
  biblio.biblionumber,
  biblio.frameworkcode,
  biblio.author,
  biblio.title,
  biblio.unititle,
  biblio.notes,
  biblio.serial,
  biblio.seriestitle,
  biblio.copyrightdate,
  biblio.timestamp,
  biblio.datecreated,
  biblio.abstract,
  Group_Concat(items.itype) AS ITYPES
FROM
  biblio
LEFT JOIN items on (biblio.biblionumber = items.biblionumber)
WHERE biblio.biblionumber = <<Enter biblio number>>
GROUP BY
  biblio.biblionumber
ORDER BY
 biblio.biblionumber
  DESC
